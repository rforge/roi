Nonlinear Programming with ROI

Example 1

The following example solves the Rosenbrock function with ROI.plugin.nloptr.

$$minimize \ f(x) = 100 \ (x_2 - x_1^2)^2 + (1 - x_1)^2$$ 
library("ROI")
## ROI: R Optimization Infrastructure
## Registered solver plugins: nlminb, ecos, glpk, ipop, quadprog, nloptr, scs, symphony.
## Default solver: auto.

f <- function(x) {
    return( 100 * (x[2] - x[1]^2)^2 + (1 - x[1])^2 )
}
f.gradient <- function(x) {
    return( c( -400 * x[1] * (x[2] - x[1] * x[1]) - 2 * (1 - x[1]),
               200 * (x[2] - x[1] * x[1])) )
}

control <- list(algorithm  = "NLOPT_GD_MLSL",
              	maxeval    = 10000,
              	population = 4,
              	local_opts = list(algorithm = "NLOPT_LD_LBFGS", xtol_rel  = 1e-4))

x <- OP( objective = F_objective(f, n=1L, G=f.gradient), 
         bounds = V_bound(li=1:2, ui=1:2, lb=c(-3, -3), ub=c(3,  3)) )

nlp <- ROI_solve(x, solver="nloptr", control, start=c(-1.2, 1))
nlp
## Optimal solution found.
## The objective value is: 1.590683e-24

solution(nlp)
## [1] 1 1

attr(nlp, "meta")
## $solver
## [1] "nloptr"
## 
## $algorithm
## [1] "NLOPT_GD_MLSL"
