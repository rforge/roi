Quadratic Programming with ROI

Example 1

library("ROI")
## ROI: R Optimization Infrastructure
## Registered solver plugins: nlminb, ecos, glpk, ipop, quadprog, scs, symphony.
## Default solver: auto.
$$\text{minimize } \ \ x_1^2 + x_2^2 + 2 x_2$$ $$\begin{array}{rrrrrr} subject \ to & x_1 & + & x_2 & = & 2 \\ \end{array} $$ $$x_1, x_2 \geq 0$$ 
qp <- OP(objective = Q_objective(2*diag(2), L=c(0, 2)),
         constraints = L_constraint(L = c(1, 1), dir = "==", rhs = 2),
         maximum = FALSE)

sol <- ROI_solve(qp)
## sol
## Optimal solution found.
## The objective value is: 3.500000e+00
solution(sol)
## [1] 1.5 0.5
